var  str ="２０１８　ｙａｅｒ　８　ｍｏｎｔｈ　２４　ｄａｙ"
var array = str.toCharArray
for (i <- 0 until  array.length){
  if (array(i) == 12288){
	array(i) = ' '
  } else if (array(i) >=  ' ' && array(i) <= 65374){
	array(i) = (array(i)-65248).toChar
  }
}
println(array.mkString)

  }
